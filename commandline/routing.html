Commandline: Routing

passing arguments

$app->cmd('mv <src> <target>', function(){
    $src = $this->params->src;
    $target = $this->params->target;
});

options

$app->set('pretty');

$app->cmd('mv <src> <target>', function(){
    return $this->params;
})->options(['overwrite','interactive']);

php cli.php mv --overwrite foo bar

{
    "overwrite": true,
    "interactive": false,
    "src": "foo",
    "target": "bar"
}

options with values

$app->set('pretty');

$app->cmd('mail <to>', function(){
    return $this->params;
})->options(['cc'=>'','subject'=>'untitled']);

php cli.php mail --cc=foo@mail.com bar@mail.com

{
    "cc": "foo@mail.com",
    "subject": "untitled",
    "to": "bar@mail.com"
}
