Database: Redis

install the redis extension phpredis

configuration

$exports['redis'] = [
    'product' => [
        'host'     => 'localhost',
        'pconnect' => true,
    ]
];

return $exports;

register as a component

$app->register('redis','\zf\Redis');

$app->get('/product/:id', function(){
    $this->redis->product->incr('view:'.$this->params->id);
});
